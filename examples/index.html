Basic Setup

import { AnalyticsEngine, BrowserContextProvider, BrowserStorageAdapter } from '@raminjafary/analytics-engine';

const engine = new AnalyticsEngine({
  maxQueueSize: 1000,
  batchTimeout: 2000,
  lazyLoading: true,
  debug: true
});

const contextProvider = BrowserContextProvider.getInstance();
engine.setContextProvider(contextProvider);

engine.addProvider(amplitudeProvider, 'amplitude');
engine.addProvider(gaProvider, 'google-analytics');

engine.send('page_view', { page: '/home' });
engine.send('user_action', { action: 'click', element: 'button' });

Custom Provider

import { BaseAnalyticsAdapter } from '@raminjafary/analytics-engine';

class CustomAnalyticsAdapter extends BaseAnalyticsAdapter {
  async load(callback) {
    await this.loadCustomSDK();
    this.isLoaded = true;
    callback?.();
  }

  init() {
    this.customSDK.init(this.apiKey);
    this.isInitialized = true;
  }

  send(eventName, options) {
    this.customSDK.track(eventName, options);
  }
}

const customProvider = new CustomAnalyticsAdapter();
engine.addProvider(customProvider, 'custom-analytics');
